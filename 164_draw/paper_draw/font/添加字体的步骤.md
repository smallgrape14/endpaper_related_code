# 1. 下载字体
https://github.com/trishume/OpenTuringCompiler/blob/master/stdlib-sfml/fonts/Times%20New%20Roman.ttf

# 2. 定位  Matplotlib 字体目录
import matplotlib
print(matplotlib.matplotlib_fname()) 

# 3. 将字体文件复制到同级目录下的 Matplotlib 所在目录的 fonts/ttf/文件夹
cp Times%20New%20Roman.ttf /home/oem/anaconda3/lib/python3.9/site-packages/matplotlib/mpl-data/fonts/ttf/

/home/oem/anaconda3/lib/python3.9/site-packages/matplotlib/mpl-data/fonts/ttf

# 4. 清除字体缓存​​删除缓存目录（路径通过以下命令获取）

#/home/oem/.cache/matplotlib
