### 描述下 File Read 的通信逻辑

#### Client 和 server 的代码逻辑
```c
// Client
1. on_pre_conn() //连接的准备工作，注册 ctx->buffer 和 ctx->msg 
    post_receive() //接收 Server 端发来的远端地址信息
// Server 
1. on_connection()
    {
        ctx->msg->id = MSG_MR;//设置 msgid 
        // msg 中放置地址信息
        ctx->msg->data.mr.addr = (uintptr_t)ctx->buffer_mr->addr;
        ctx->msg->data.mr.rkey = ctx->buffer_mr->rkey;
        // 发送本地的地址信息给 client 端，即告知 client file path 写入这里
        send_message(id);
    }
// Client
2. on_completion()
    (1)  if (ctx->msg->id == MSG_MR) {// 1th 发送请求的文件路径信息
        // 1. 接收消息中的远端 buffer 信息，设置 peer_addr
        ctx->peer_addr = ctx->msg->data.mr.addr;//对方的buffer地址
        ctx->peer_rkey = ctx->msg->data.mr.rkey;//对方的buffer rkey
        // 2. RDMA Write 把 file_path 信息写入到上面的远端buffer 中
        printf("received MR, sending file name\n");
        send_file_name(id);
            ==> write_remote()
                {
                    // 2.1 操作码 RDMA IMM Write,设置 imm_data 存放数据长度，让接收端可以识别不同的消息类型
                    wr.opcode = IBV_WR_RDMA_WRITE_WITH_IMM;//RDMA写操作
                    wr.imm_data = htonl(len);//PS: 这里的len是文件数据的长度

                    // 2.2 file_path 写入远端的地址
                    wr.wr.rdma.remote_addr = ctx->peer_addr;//对方的buffer地址
                    wr.wr.rdma.rkey = ctx->peer_rkey;//对方的buffer rkey
                    // 2.3 本地内存地址 存放要传输的 file_path
                    sge.addr = (uintptr_t)ctx->buffer;
                    sge.length = len;
                    sge.lkey = ctx->buffer_mr->lkey;
                    // 2.4 发送 
                    ibv_post_send()
                }            
         }
// Server
2. on_completion()
{
    if (wc->opcode == IBV_WC_RECV_RDMA_WITH_IMM) {
        if ( size )
        {
            // 2.1 file_name copy to ctx->buffer
            memcpy(ctx->file_name, ctx->buffer, size);
            // 2.2 open file / read file to ctx->buffer
            ctx->fd = open(ctx->file_name,O_RDONLY);
            long size = getFileSize(ctx->file_name);
            int read_res = read(ctx->fd, ctx->buffer, BUFFER_SIZE<size?BUFFER_SIZE:size);
            // 2.3 通知对方可以开始接收文件数据了,通过设置 msg 状态
            ctx->msg->id = MSG_READY;
            send_message()
                ==> ibv_post_send()
        }
    }
}
// Client
    (2) else if (ctx->msg->id == MSG_READY) {// 2th RDMA Read 接收文件数据
        printf("received READY, sending chunk\n");
        // send_next_chunk(id);
        uint32_t read_length = 16; // 读取的数据长度
        const char *output_file = "output.txt"; // 输出文件路径
        read_remote(id, read_length, output_file);
            ==> {   
                // 2.1 设置操作码
                    wr.opcode = IBV_WR_RDMA_READ; // RDMA Read 操作 
                // 2.2 对端存放 file data 的内存地址信息
                    wr.wr.rdma.remote_addr = ctx->peer_addr; // 对端的 buffer 地址
                    wr.wr.rdma.rkey = ctx->peer_rkey;        // 对端的 buffer rkey  
                // 2.3 本地内存地址信息，存放 RDMA Read 到的文件数据
                    sge.addr = (uintptr_t)ctx->buffer; // 本地缓冲区地址，用于存储读取的数据
                    sge.length = len;                  // 读取的数据长度
                    sge.lkey = ctx->buffer_mr->lkey;   // 本地缓冲区的 lkey
                // 2.4 提交请求
                    ibv_post_send()
                // 2.5 将读取的数据从 ctx->buffer 写入本地文件
                    fwrite(ctx->buffer, 1, len, file);
                }
    }
//Server
```

### reference

#### MSG_MR
MSG_MR 是一个消息类型，用于在 RDMA 通信中交换内存区域（Memory Region, MR）的信息。具体来说，MSG_MR 消息包含了对端需要进行 RDMA 操作的内存区域的元数据，如内存地址、长度、本地密钥（lkey）和远程密钥（rkey）等。
作用
**交换内存区域信息**：在 RDMA 通信中，双方需要知道对方的内存区域信息，以便进行高效的内存读写操作。MSG_MR 消息用于在连接建立后，将本地的内存区域信息发送给对端，对端收到后可以使用这些信息来执行 RDMA 操作。
**触发 RDMA 操作**：当一方收到 MSG_MR 消息后，可以根据消息中的内存区域信息，构建并发送 RDMA 读写请求，从而实现高效的内存操作
